void setup() {
  
  for(int i=0; i<64; i++)                     //100 사이즈의 int 자료형 배열을 만든 다음에 그 배열의 첫 64개 요소들을 0으로 초기화
  {
    data[i]=0;
  }
  pinMode(ledPower,OUTPUT);   
  
  Serial.begin(9600);                         //9600 보드레이트로 시리얼 통신 시작

  PMS7003.begin(9600);                        //미세먼지 센서와 통신 시작
  
  lcd.init();                                 //LCD 초기화
  lcd.backlight();                            //LCD 백라이트 켜기
  lcd.setCursor(0,0);                         //LCD에 글자를 1행 1열부터 글씨를 출력
  lcd.clear();                                //LCD의 글자를 모두 지움
  lcd.print("starting...");                   //"Starting"출력(LCD)
  delay(500);                                 //0.05초 대기
  
  Serial.println("starting...");              //"Starting"출력(시리얼 모니터)
  delay(500);
  
  lcd.setCursor(0,0);
  
    
  if(!ccs.begin()){                           //TVOC 센서가 시작되지 않았을 때
    lcd.clear();
    lcd.print("Failed to start");
    Serial.print("Failed to start");
    while(1);
  lcd.setCursor(0,1);
  lcd.print("sensor starting");
  delay(500);
  //hcho.begin();
  }
  lcd.setCursor(12,0);
  while(!ccs.available());                    //
  lcd.print(".");
  delay(500);
}
    
void loop() {

  //---------------------미세먼지센서---------------------------
  if(PMS7003.available());                    //미세먼지 센서가 작동할 때
  {
    data[count] = PMS7003.read();             //센서 측정값을 data에 저장
    count=count+1;

    if (count < 64)
    {
      return;
    }
   }
  
 // lcd.print(count);
  if(count>=64)
  {
    PMS7003.end();  
    for (int i=1;i<33;i++)
    {
      check=0;
      for(int j=i;j<i+30;j++)
      {
        check=check+data[j];
      }
      if(data[i]==66&data[i+1]==77&(data[i+30]*16*16+data[i+31])==check)    //오류가 없는지 확인
      {
      /*Serial.print("PM1.0 = ");
      a=data[i+10]*256+data[i+11];
      Serial.print(a);
      Serial.print("   ");*/
      Serial.print("PM2.5 = ");   
      b=data[i+12]*256+data[i+13];             //값을 단위에 맞게 변환
      Serial.print(b);
      Serial.print("   ");
      Serial.print("PM10 = ");
      c=data[i+14]*256+data[i+15];             //값을 단위에 맞게 변환
      Serial.print(c);
      Serial.println("");
      lcd.clear();
      lcd.setCursor(0,0);
      //lcd.print("PM1.0 = ");
      //lcd.print(a);
      //lcd.setCursor(0,1);
      lcd.print("PM2.5 = ");                   //LCD에 출력
      lcd.print(b);
      lcd.setCursor(0,1);
      lcd.print("PM10 = ");
      lcd.print(c);

      }
      else {
        lcd.clear();
        lcd.print("ERROR(PMS7003)");          //오류가 발생했을 때
      }
    }

    count=1;
    delay(500);
    PMS7003.begin(9600);   
  }
  delay(1500);                           
  //----------------------TVOC센서----------------------------
  lcd.setCursor(0,0);
  lcd.clear();
  
  if(ccs.available()){                       //TVOC센서가 작동할 때
    if(!ccs.readData()){
      lcd.print("CO2: ");   
      lcd.print(ccs.geteCO2());              //LCD에 CO2 측정값을 출력
      lcd.print("ppm");
      lcd.setCursor(0,1);
      lcd.print("TVOC:");
      lcd.print(ccs.getTVOC());              //LCD에 TVOC 측정값을 출력
      lcd.print("ppb");
      Serial.print("CO2: ");
      Serial.print(ccs.geteCO2());
      Serial.print("ppm, TVOC: ");
      Serial.println(ccs.getTVOC());
      delay(2000);
    }
    else{                                   //오류가 발생했을 때
      lcd.print("ERROR(TVOC)-");
      lcd.setCursor(0,1);
      lcd.print("CannotReadData");
      Serial.println("😢CannotReadData");
      //while(1);
      delay(700);
    }
  }
  //----------------------hcho센서----------------------------
  float vol_Formol;
  float ppm_Formol;

  sensorValue=analogRead(A1);               //A1 핀 출력값을 변수에 저장

  lcd.clear();
  lcd.setCursor(0,0);
  vol_Formol=(float)sensorValue/1024*1.2;   //단위에 맞게 변환
  ppm_Formol=(float)vol_Formol*(22.4/30);
  lcd.print("Gas:");                        //LCD에 출력
  lcd.print(vol_Formol);                    
  lcd.print("mg/m^3");
  Serial.print("Gas:");
  Serial.print(vol_Formol);
  Serial.println("mg/m³");
  lcd.setCursor(0,1);
  lcd.print("Gas:");
  lcd.print(ppm_Formol);
  lcd.print("PPM");
  Serial.print("Gas:");
  Serial.print(ppm_Formol);
  Serial.println("PPM");
  /*hcho.read();
  //lcd.print(hcho.read());
  hcho.display();*/
  delay(2000);
  lcd.clear();
}
